```typescript
import { test, expect } from '@playwright/test';

test('Allow users to reset password via "Forgot Password" link', async ({ page }) => {
    // Navigate to the login page
    await page.goto('https://yourwebsite.com/login');

    // Click on the "Forgot Password" link
    await page.click('text=Forgot Password');

    // Verify redirection to password reset form
    await expect(page).toHaveURL('https://yourwebsite.com/reset-password');

    // Enter registered email and submit
    await page.fill('input[name=email]', 'test@example.com');
    await page.click('button[type=submit]');

    // Verify email with reset instructions
    const resetEmail = await page.waitForSelector('emailSelector');
    expect(resetEmail).not.toBeNull();

    // If email not registered, verify error message
    const errorMessage = await page.waitForSelector('errorSelector');
    expect(errorMessage).not.toBeNull();

    // Click on the reset link in the email
    await page.click('emailLink');

    // Set a new password that meets complexity requirements
    await page.fill('input[name=newPassword]', 'NewPassword123');
    await page.fill('input[name=confirmPassword]', 'NewPassword123');
    await page.click('button[type=submit]');

    // Verify successful password reset
    const successMessage = await page.waitForSelector('successMessageSelector');
    expect(successMessage).not.toBeNull();

    // Additional checks
    // Make sure feature is accessible from both desktop and mobile
    // Verify no password is shown in plain text
});
```
